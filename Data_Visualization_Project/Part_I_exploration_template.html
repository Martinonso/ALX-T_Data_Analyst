def plot_chart(df,x,y,t,k):
    df.plot(y=y, x=x, title=t, rot=90, kind=k, figsize=(10,10))
    plt.xlabel(x, fontsize=12)
    plt.ylabel(y, fontsize=12)
    return plt.show()

def bar_chart(df,x,y,h,t):
    plt.figure(figsize = [10, 10])
    sb.countplot(data=df, x=x, hue=h).set(xlabel=x, ylabel=y, title=t)
    plt.xlabel(x, fontsize=12)
    plt.ylabel(y, fontsize=12)
    plt.title(t, fontsize=12)
    plt.legend()
    return plt.show();

#creating some functions that will be used to plot our chart
def box_chart(df,x,y,h,t):
    plt.figure(figsize = [10, 10])
    sb.boxplot(data=df, x=x, y=y, hue=h).set(xlabel=x, ylabel=y, title=t)
    plt.xlabel(x, fontsize=12)
    plt.ylabel(y, fontsize=12)
    plt.title(t, fontsize=12)
    plt.legend()
    return plt.show();

def violin_chart(df,x,y,h,t):
    plt.figure(figsize = [10, 10])
    sb.violinplot(data=df, x=x, y=y, hue=h).set(xlabel=x, ylabel=y, title=t)
    plt.xlabel(x, fontsize=12)
    plt.ylabel(y, fontsize=12)
    plt.title(t, fontsize=12)
    plt.legend()
    return plt.show();

def bar_chart(df,x,h,t):
    plt.figure(figsize = [10, 10])
    sb.countplot(data = df, x = x, hue = h).set(xlabel=x, title=t)
    plt.xlabel(x, fontsize=12)
    plt.title(t, fontsize=12)
    plt.legend()
    return plt.show();

def scatter_chart(df,x,y,t):
    plt.figure(figsize = [10, 10])
    sb.regplot(data = df, x = x, y = y).set(xlabel=x, ylabel=y, title=t)
    plt.xlabel(x, fontsize=12)
    plt.ylabel(y, fontsize=12)
    plt.title(t, fontsize=12)
    plt.legend()
    return plt.show();

def stacked_chart(df,x,h,p,q,t):
    plt.figure(figsize = [10, 10])
    sb.countplot(data = df, x = x, hue = h, order = p, hue_order = q).set(xlabel=x, ylabel=y, title=t)
    plt.xlabel(x, fontsize=12)
    plt.ylabel(y, fontsize=12)
    plt.title(t, fontsize=12)
    plt.legend()
    return plt.show();

We have less female participants that shared their by bike but more aged people when compared with those of their male counterparts.
